import java.io.PrintWriter;
import java.util.Random;

import org.apache.commons.math.stat.descriptive.moment.FirstMoment;
import org.apache.commons.math.stat.descriptive.moment.FourthMoment;
import org.apache.commons.math.stat.descriptive.moment.Kurtosis;
import org.apache.commons.math.stat.descriptive.moment.Mean;
import org.apache.commons.math.stat.descriptive.moment.SecondMoment;
import org.apache.commons.math.stat.descriptive.moment.Skewness;
import org.apache.commons.math.stat.descriptive.moment.StandardDeviation;
import org.apache.commons.math.stat.descriptive.moment.ThirdMoment;
import org.apache.commons.math.stat.descriptive.moment.Variance;

/**
 * This is the class used to generate the statistics-moment.txt data. The values
 * where calculated by the Apache Commons Math library:
 * http://commons.apache.org/math/
 *
 * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
 * @version $Id: MeanTest.java 707 2010-12-16 18:37:37Z fwilhelm $
 */
public class StatisticMoments {

	public static void main(String[] args) throws Exception {
		final Random random = new Random(1234567890);
		
		final FirstMoment m1 = new FirstMoment();
		final SecondMoment m2 = new SecondMoment();
		final ThirdMoment m3 = new ThirdMoment();
		final FourthMoment m4 = new FourthMoment();
		final Mean mean = new Mean();
		final Variance variance = new Variance();
		final StandardDeviation std = new StandardDeviation();
		final Kurtosis kurtosis = new Kurtosis();
		final Skewness skewness = new Skewness();
		
		final PrintWriter writer = new PrintWriter("statistic-moments.txt");
		writer.println(String.format(
				"# %s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s",
				"Number",
				"First monent",
				"Second moment",
				"Third moment",
				"Fourth moment",
				"Mean",
				"Variance",
				"Std",
				"Kurtosis",
				"Skewness"
			));
		
		
		for (int i = 0; i < 750; ++i) {
			final double number = random.nextDouble()*100000 - 50000;
			
			m1.increment(number);
			m2.increment(number);
			m3.increment(number);
			m4.increment(number);
			mean.increment(number);
			variance.increment(number);
			std.increment(number);
			kurtosis.increment(number);
			skewness.increment(number);
			
			String line = String.format(
					"%.20f\t%.20f\t%.20f\t%.20f\t%.20f\t%.20f\t%.20f\t%.20f\t%.20f\t%.20f", 
					number, 
					m1.getResult(),
					m2.getResult(),
					m3.getResult(),
					m4.getResult(),
					mean.getResult(),
					variance.getResult(),
					std.getResult(),
					kurtosis.getResult(),
					skewness.getResult()
				);
			
			writer.println(line);
		}
		
		writer.close();
	}
	
}
